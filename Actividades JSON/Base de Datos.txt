create database if not exists dc;

use dc;

create table if not exists treballs (
	id int primary key not null auto_increment, 
	nombre varchar(255)
);

create table if not exists personas (
	nombre varchar(255),
    	treball int,
    	horari varchar(255),
    	email varchar(255),
    	foreign key (treball) references treballs(id)
);
